We will add another menu item for Elementor Pro resources. It'll have to be along the other two menu items under Tools on the left sidebar. This can become an organization nightmare.

You'd think of adding a submenu inside the Tools submenu, but Wordpress' menu system is **only two levels deep**.

We can instead add a separator where a third menu item could be. By NOT providing a link, page, or callback, you can add a separator that visually break up groups

We have a root menu item already "WordPress Resources". When creating that root level menu item, we gave it a menu slug `wf-wordpress-resources`. 

Now instead of adding submenus to a "tools.php", add it to that menu slug that's at the root level: "wf-wordpress-resources". So your submenu items are changed to:
```
        // Gutenberg Resources
        add_submenu_page(
            'wf-wordpress-resources',                           // Parent slug (your top-level menu)
            __('Gutenberg Resources', 'wf-gutenberg-resources'),

            'manage_options',
            'wf-gutenberg-resources',
```

```
/**

    exit;
}

class WF_Resources {

    public function __construct() {
        add_action('admin_menu', array($this, 'add_admin_menu'));
        add_action('admin_menu', array($this, 'add_submenus'));
    }

    /**
     * Add a top-level menu in the admin sidebar
     */
    public function add_admin_menu() {
        add_menu_page(
            __('WordPress Resources', 'wf-wordpress-resources'), // Page title
            __('WordPress Resources', 'wf-wordpress-resources'), // Menu title
            'manage_options',                                   // Capability
            'wf-wordpress-resources',                           // Menu slug
            array($this, 'render_resources_page'),              // Callback
            'dashicons-wordpress-alt',                          // Icon
            25                                                  // Position
        );
    }

    /**
     * Add submenus under the WordPress Resources top-level menu
     */
    public function add_submenus() {
        // Gutenberg Resources
        add_submenu_page(
            'wf-wordpress-resources',                           // Parent slug (your top-level menu)
            __('Gutenberg Resources', 'wf-gutenberg-resources'),
            __('Gutenberg Resources', 'wf-gutenberg-resources'),
            'manage_options',
            'wf-gutenberg-resources',
            array($this, 'render_gutenberg_page')
        );

        // WPBakery Resources
        add_submenu_page(
            'wf-wordpress-resources',
            __('WPBakery Resources', 'wf-wpbakery-resources'),
            __('WPBakery Resources', 'wf-wpbakery-resources'),
            'manage_options',
            'wf-wpbakery-resources',
            array($this, 'render_wpbakery_page')
        );

        // Example future: Elementor Resources
        add_submenu_page(
            'wf-wordpress-resources',
            __('Elementor Resources', 'wf-elementor-resources'),
            __('Elementor Resources', 'wf-elementor-resources'),
            'manage_options',
            'wf-elementor-resources',
            array($this, 'render_elementor_page')
        );
    }

    // Top-level main page
    public function render_resources_page() {
        ?>
        <div class="wrap">
            <h1><?php esc_html_e('WordPress Resources', 'wf-wordpress-resources'); ?></h1>
            <p>Here are some helpful WordPress resources:</p>
            <ul>
                <li><a href="https://developer.wordpress.org/" target="_blank">WordPress Developer Resources</a></li>
                <li><a href="https://wordpress.org/support/" target="_blank">WordPress Support Forums</a></li>
                <li><a href="https://make.wordpress.org/" target="_blank">Make WordPress (Contributor Portal)</a></li>
            </ul>
        </div>
        <?php
    }

    // Gutenberg submenu page
    public function render_gutenberg_page() {
        ?>
        <div class="wrap">
            <h1><?php esc_html_e('Gutenberg Resources', 'wf-gutenberg-resources'); ?></h1>
            <p>Here are some helpful Gutenberg resources:</p>

                <li><a href="https://developer.wordpress.org/block-editor/" target="_blank">Block Editor Handbook</a></li>

                <li><a href="https://github.com/WordPress/gutenberg" target="_blank">Gutenberg GitHub Repository</a></li>
            </ul>
        </div>
        <?php
    }

    // WPBakery submenu page
    public function render_wpbakery_page() {
        ?>
        <div class="wrap">

    }
}

 * Initialize the plugin
 */
function wff_add_resources() {
    new WF_Resources();

add_action('plugins_loaded', 'wff_add_resources');
```

The Azure CLI (`az`) is the main command-line tool for managing Azure resources. This lesson is a reference sheet for the commands used most often.

---

# 1. Install and Authenticate

### Install

```bash
# macOS
brew update && brew install azure-cli

# Windows
winget install Microsoft.AzureCLI

# Linux (Ubuntu/Debian)
curl -sL https://aka.ms/InstallAzureCLIDeb | sudo bash
```

### Sign in

```bash
az login                              # opens browser for interactive login
az login --use-device-code            # useful in headless/SSH sessions
```

### Select subscription

```bash
az account list --output table        # list all subscriptions
az account show                       # show current subscription
az account set --subscription <name or ID>
```

# 2. Resource Groups

```bash
# Create
az group create --name rg-myproject --location eastus

# List
az group list --output table

# Show
az group show --name rg-myproject

# Delete (also deletes everything inside)
az group delete --name rg-myproject --yes --no-wait
```

# 3. Storage

```bash
# Create storage account
az storage account create \
  --name mystorageaccount \
  --resource-group $RG \
  --location $LOC \
  --sku Standard_LRS

# List storage accounts
az storage account list --resource-group $RG --output table

# Get account key
az storage account keys list -g $RG -n $ACCOUNT --query "[0].value" -o tsv

# Create container
az storage container create \
  --account-name $ACCOUNT \
  --name mycontainer \
  --account-key $KEY

# Upload blob
az storage blob upload \
  --account-name $ACCOUNT --account-key $KEY \
  --container-name mycontainer \
  --name myfile.txt --file ./myfile.txt

# List blobs
az storage blob list \
  --account-name $ACCOUNT --account-key $KEY \
  --container-name mycontainer --output table

# Download blob
az storage blob download \
  --account-name $ACCOUNT --account-key $KEY \
  --container-name mycontainer \
  --name myfile.txt --file ./downloaded.txt

# Delete blob
az storage blob delete \
  --account-name $ACCOUNT --account-key $KEY \
  --container-name mycontainer --name myfile.txt

# Generate SAS URL (readable for 1 hour)
END=$(date -u -d "+1 hour" '+%Y-%m-%dT%H:%MZ')   # Linux
# END=$(date -u -v+1H '+%Y-%m-%dT%H:%MZ')         # macOS

az storage blob generate-sas \
  --account-name $ACCOUNT --account-key $KEY \
  --container-name mycontainer --name myfile.txt \
  --permissions r --expiry $END --full-uri --output tsv
```

# 4. Virtual Machines

```bash
# Create Linux VM
az vm create \
  --resource-group $RG \
  --name $VM \
  --image Ubuntu2204 \
  --size Standard_B1s \
  --admin-username azureuser \
  --generate-ssh-keys

# Show public IP
az vm show -g $RG -n $VM --show-details --query publicIps -o tsv

# SSH shortcut
ssh azureuser@$(az vm show -g $RG -n $VM --show-details --query publicIps -o tsv)

# Stop (still billed)
az vm stop -g $RG -n $VM

# Deallocate (stops compute billing)
az vm deallocate -g $RG -n $VM

# Start
az vm start -g $RG -n $VM

# List VMs
az vm list -g $RG --output table

# Open port
az vm open-port -g $RG -n $VM --port 80

# Delete
az vm delete -g $RG -n $VM --yes
```

# 5. App Service

```bash
# Create plan
az appservice plan create \
  --name $PLAN --resource-group $RG \
  --sku F1 --is-linux

# Create web app
az webapp create \
  --name $APP --resource-group $RG \
  --plan $PLAN --runtime "NODE:20-lts"

# Deploy zip
az webapp deploy \
  --resource-group $RG --name $APP \
  --src-path app.zip --type zip

# Set app settings
az webapp config appsettings set \
  --resource-group $RG --name $APP \
  --settings KEY=value

# Stream logs
az webapp log tail --resource-group $RG --name $APP

# Restart
az webapp restart --resource-group $RG --name $APP
```

# 6. Azure Functions

```bash
# Create function app
az functionapp create \
  --name $FUNCAPP --resource-group $RG \
  --storage-account $STORAGE \
  --consumption-plan-location $LOC \
  --runtime node --runtime-version 20 \
  --functions-version 4 --os-type Linux

# Set app settings
az functionapp config appsettings set \
  --name $FUNCAPP --resource-group $RG \
  --settings KEY=value

# Get function keys
az functionapp keys list --name $FUNCAPP --resource-group $RG

# Stream logs
az functionapp log tail --name $FUNCAPP --resource-group $RG
```

# 7. SQL Database

```bash
# Create server
az sql server create \
  --name $SERVER --resource-group $RG \
  --location $LOC \
  --admin-user sqladmin \
  --admin-password "YourStr0ng!Pass"

# Create database (serverless)
az sql db create \
  --server $SERVER --resource-group $RG \
  --name $DB \
  --edition GeneralPurpose \
  --compute-model Serverless \
  --family Gen5 --capacity 1 \
  --auto-pause-delay 60

# Add firewall rule for your IP
az sql server firewall-rule create \
  --resource-group $RG --server $SERVER \
  --name AllowMyIP \
  --start-ip-address $(curl -s ifconfig.me) \
  --end-ip-address $(curl -s ifconfig.me)

# Connection string
az sql db show-connection-string \
  --server $SERVER --name $DB --client ado.net
```

# 8. Key Vault

```bash
# Create vault
az keyvault create \
  --name $VAULT --resource-group $RG \
  --location $LOC

# Add secret
az keyvault secret set \
  --vault-name $VAULT --name SecretName --value "secretvalue"

# Read secret
az keyvault secret show \
  --vault-name $VAULT --name SecretName \
  --query value -o tsv

# List secrets
az keyvault secret list --vault-name $VAULT --output table

# Grant RBAC access
az role assignment create \
  --assignee $OID \
  --role "Key Vault Secrets User" \
  --scope $(az keyvault show --name $VAULT --query id -o tsv)
```

# 9. RBAC and Identities

```bash
# List role assignments for a resource group
az role assignment list --resource-group $RG --output table

# Assign role to user
az role assignment create \
  --assignee alice@example.com \
  --role "Contributor" \
  --resource-group $RG

# Enable managed identity on App Service
az webapp identity assign --resource-group $RG --name $APP

# Get managed identity object ID
az webapp identity show --resource-group $RG --name $APP \
  --query principalId -o tsv

# List available roles
az role definition list --query "[].{Name:roleName}" --output table
```

# 10. Useful Global Options

```bash
# Format output
az ... --output table    # table
az ... --output json     # full JSON
az ... --output tsv      # tab-separated (good for scripting)
az ... --output yaml     # YAML

# Query with JMESPath
az vm show -g $RG -n $VM --query "osProfile.computerName" -o tsv

# Dry run / what-if
az deployment group what-if ...

# Verbose output for debugging
az ... --verbose

# Debug mode
az ... --debug
```

# 11. Set Default Resource Group and Location

Avoid typing `--resource-group` and `--location` every time:

```bash
az configure --defaults group=rg-myproject location=eastus
```

Now `az webapp create --name myapp ...` uses those defaults.

Clear defaults:

```bash
az configure --defaults group='' location=''
```

# 12. Useful Shortcuts

```bash
# Show current logged-in user
az account show --query user.name -o tsv

# List all locations
az account list-locations --output table

# Find a VM image
az vm image list --publisher Canonical --sku 22_04-lts-gen2 --all --output table

# Get subscription ID
az account show --query id -o tsv

# What version of CLI is installed
az version
```
